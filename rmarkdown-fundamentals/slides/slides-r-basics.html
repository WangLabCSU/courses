自定义函数

基本结构

# 定义函数
calculate_bmi <- function(weight_kg, height_m) {
  bmi <- weight_kg / (height_m ^ 2)
  
  # 分类
  category <- ifelse(bmi < 18.5, "偏瘦",
              ifelse(bmi < 25, "正常",
              ifelse(bmi < 30, "超重", "肥胖")))
  
  return(list(bmi = round(bmi, 2), category = category))
}

# 调用函数
calculate_bmi(70, 1.75)
$bmi
[1] 22.86

$category
[1] "正常"

函数参数

# 默认参数
analyze_gene <- function(expression, 
                         gene_name = "Unknown",
                         threshold = 10) {
  mean_exp <- mean(expression)
  is_expressed <- mean_exp > threshold
  
  list(
    gene = gene_name,
    mean = mean_exp,
    expressed = is_expressed
  )
}

# 使用
expr <- c(5, 8, 12, 15, 20)
analyze_gene(expr, gene_name = "TP53", threshold = 8)
$gene
[1] "TP53"

$mean
[1] 12

$expressed
[1] TRUE
